use std::io::{stdin, stdout, stdout_mut};

fn main() {
    let mut input = String::new();
    println!("Enter a Rust string:");
    stdin().read_line(&mut input).expect("Failed to read line");
    let input: &str = input.trim();
    let mut output = String::new();
    let mut count = 0;
    for c in input.chars() {
        if c.is_ascii_punct() {
            if c.is_ascii_vowel() {
                count += 1;
            }
        }
    }
    println!("The number of ASCII vowel characters in the input string is: {}", count);
}
