use std::io::{stdin, stdout, stdout::flush, stdout::write};

fn main() {
    let mut input = String::new();
    let mut output = String::new();

    // Read input from stdin
    stdout().flush().expect("Failed to flush stdout");
    stdin().read_line(&mut input).expect("Failed to read input");

    // Split input into words
    let words = input.split_whitespace().collect::<Vec<_>>();

    // Count vowel characters
    let count = words.iter().filter(|w| w.chars().all(|c| c.is_ascii_punct() || c.is_ascii_vowel()))
        .count();

    // Print result
    println!("Vowel count: {}", count);

    // Write output to stdout
    stdout().write_all(output.as_bytes()).expect("Failed to write output");
}
